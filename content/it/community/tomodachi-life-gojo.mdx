### Passi per Promuovere Relazioni Ispirate a JJK

<Steps>
  <Step num="1" title="Amicizie Iniziali">

  </Step>
  <Step num="2" title="Introduci Conflitti/Drammi">

  </Step>

  </Step>
  <Step num="4" title="Monitora la Compatibilità">

  </Step>
</Steps>
