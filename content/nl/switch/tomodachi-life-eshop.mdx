<Section icon={Target} title="Hoe krijg je toegang tot de Living the Dream Demo">

  <Step num="1" title="Toegang tot de Nintendo eShop">

  </Step>
  <Step num="2" title="Zoek naar het spel">

  </Step>
  <Step num="3" title="Download de demo">

  </Step>
  <Step num="4" title="Speel de demo">

</Steps>
